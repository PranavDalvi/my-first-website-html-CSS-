import random
start = 1 # starting point of dice
stop = 6 # ending point of dice
            
roll_again = 'Y'
            
while roll_again == 'Y':
    values = random.randint(start,stop)
    print(f"The values are: {values} ")
    roll_again = input("Do you want to roll again? [y or n]")[0].upper()
else:
    print('Thank you')
